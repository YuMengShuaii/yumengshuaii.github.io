LoadRunner压测Get方法

发表于 2017-03-27 |

  • 首先LoadRunner本身具有很多方法均可测试接口，今天我们使用web_custom_request方法进行Get接口测试。

    int nHttpRetCode;//设置http请求code  
web_set_max_html_param_len("262144"); // 默认最大长度为256 
web_reg_save_param("ResponseBody","LB=","RB=","Search=Body", LAST);//设置ResponseBody
web_save_header(RESPONSE,"ResponseHeader");//添加header

  • 如有需要可以开启事务

    lr_start_transaction("Communicate");//开启事务，名为Communicate。
lr_end_transaction("Communicate", LR_PASS); //结束事务

  • 调用request方法

    web_custom_request("getname",  
           "URL=http://www.javamall.com.cn/xxx/xx/xx/xx/xx?username=xx&password=xxx",  
           "Method=GET",  
           "Resource=0",  
           "Referer=",  
           "Mode=HTTP",  
           "EncType=text/html;charset=UTF-8",  
           "Body=",  
            LAST);

  • 对返回的Json数据进行转码

    //转码
lr_convert_string_encoding(lr_eval_string("{ResponseBody}"),"utf-8",NULL,"ResponseBody");
//打印返回数据
lr_output_message("# 响应头信息：\n %s",lr_eval_string("{ResponseHeader}"));
lr_output_message("# 响应内容体：\n %s",lr_eval_string("{ResponseBody}"));

  • 获取Http请求码，并且判断是否成功！

    //获取请求码
nHttpRetCode = web_get_int_property(HTTP_INFO_RETURN_CODE);  
    //判断是否请求成功!(当请求码为200时，请求成功！)
    if(nHttpRetCode == 200)  
    {  
        lr_output_message("-------------Success!------------");  
    }  
    else  
    {   
        lr_output_message("----------------Failed!------------");  
    }

  • 完整代码

     Action() {

int nHttpRetCode;  
web_set_max_html_param_len("262144"); // 默认最大长度为256  

web_reg_save_param("ResponseBody", "LB=", "RB=", "Search=Body", LAST);  
web_save_header(RESPONSE,"ResponseHeader");  

lr_start_transaction("Communicate");  
web_custom_request("getname",  
                   "URL=http://www.javamall.com.cn/xxx/xx/xx/xx/xxx?username=xxx&password=xxx",  
                   "Method=GET",  
                   "Resource=0",  
                   "Referer=",  
                   "Mode=HTTP",  
                   "EncType=text/html;charset=UTF-8",  
                   "Body=",  
                   LAST); 

//打印返回信息
lr_convert_string_encoding(lr_eval_string("{ResponseBody}"),"utf-8",NULL,"ResponseBody");
lr_output_message("# 响应头信息：\n %s", lr_eval_string("{ResponseHeader}"));  
lr_output_message("# 响应内容体：\n %s", lr_eval_string("{ResponseBody}"));  
lr_end_transaction("Communicate", LR_PASS);  

nHttpRetCode = web_get_int_property(HTTP_INFO_RETURN_CODE);  

if(nHttpRetCode == 200)  
{  
    lr_output_message("---------------Success!------------");  
}  
else  
{  
    lr_output_message("----------------Failed!------------");  
 }  
    return 0;
}
